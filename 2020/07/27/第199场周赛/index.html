T1：重新排列字符串

思路：略

一个稍微复杂点的解法就是，构造一个Map => $indices[i]:s[i]$ 然后根据key进行排序

class Solution {
    public String restoreString(String s, int[] indices) {
        char[] chs = new char[s.length()];
        for (int i = 0; i < s.length(); i++) {
            chs[indices[i]] = s.charAt(i);
        }
        return new String(chs);
    }
}

c++

bool cmp(pair<int, char> p1, pair<int, char> p2) {
    return p1.first < p2.first;
}

class Solution {
public:
    string restoreString(string s, vector<int>& indices) {
        vector<pair<int, char>> ve;
        for (int i = 0; i < indices.size(); i++) {
            ve.push_back({indices[i], s[i]});
        }
        sort(ve.begin(), ve.end(), cmp);
        string res = "";
        for (auto p : ve) {
            res.push_back(p.second);
        }
        return res;
    }
};

T2：灯泡开关 IV

对于 n 个连续的灯泡。我们可以有一下操作，反转 i 到 n-1 个连续的灯泡

所以对于连续的1或者0，我们只有一次操作即可。统计相邻灯泡状态不同的个数即为反转次数

class Solution:
    def minFlips(self, target: str) -> int:
        s = "0" + target
        res = 0
        for i in range(1, len(s)):
            if s[i] != s[i-1]:
                res += 1
        return res

T3：好叶子节点对的数量

本质上是LCA，但是数据量很弱，所以直接暴力循环即可。统计当前节点下，所有的叶子节点。然后再循环处理，小于等于distance的情况。

class Solution {
    public int countPairs(TreeNode root, int distance) {
        if (root == null) return 0;
        List<Integer> left = new ArrayList<>();
        dfs(root.left, 0, left);
        List<Integer> right = new ArrayList<>();
        dfs(root.right, 0, right);
        int res = 0;
        for (Integer l : left) {
            for (Integer r : right) {
                // 自身算一
                if (l + r + 2 <= distance) {
                    res++;
                }
            }
        }
        // 不是到根节点，md
        res += countPairs(root.left, distance);
        res += countPairs(root.right, distance);
        return res;
    }

    private boolean isLeaf(TreeNode root) {
        return root.left == null && root.right == null;
    }

    private void dfs(TreeNode root, int depth, List<Integer> list) {
        if (root == null) return;
        if (isLeaf(root)) {
            list.add(depth);
            return;
        }
        dfs(root.left, depth + 1, list);
        dfs(root.right, depth + 1, list);
    }
}
